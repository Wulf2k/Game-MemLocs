function Generate-ChecksumUpperCase {
    param (
        [string]$str,
        $initialValue
    )

    $checksum = $initialValue
	
    foreach ($char in $str.ToCharArray()) {
        $charValue = $char
        if ($charValue -ge 97 -and $charValue -le 122) {
            $charValue = ($charValue - 97 + 65) * 0x1000000 -shr 0x18
        }
        $checksum = ($checksum * 0x7F + $charValue) % 0x100000000
    }
	#Write-Host $str "=" ([uint32]($checksum % 0x100000000)).tostring("X")
	Write-Host $str "=" ([uint32]($checksum)).tostring("X")

}
